create schema LibraryManagement;create table authors(      author_id int primary key auto_increment,      author_name varchar(255)      );create table genres(      genre_id int primary key auto_increment,      genre_name varchar(255));create table books (	book_id INT auto_increment PRIMARY KEY,	title VARCHAR(255),	publication_year YEAR,	author_id INT, 	genre_id INT,     FOREIGN KEY(genre_id) references genres(genre_id),    FOREIGN KEY(author_id) references authors(author_id));create table users(	user_id INT auto_increment PRIMARY KEY,	username VARCHAR(255),	genre_id INT,	email VARCHAR(255),    	FOREIGN KEY(genre_id) references genres(genre_id)      );create table borrowed_books(    borrow_id INT auto_increment PRIMARY KEY,    book_id INT,     user_id INT,     borrow_date DATE,    return_date DATE,    FOREIGN KEY (book_id) references books(book_id),    FOREIGN KEY (user_id) references users(user_id)    );
